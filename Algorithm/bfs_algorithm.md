#include <iostream>
#include <vector>
#include <queue>

using namespace std;

vector<vector<int>> graph = {
    {},         // 0λ² λ…Έλ“λ” μ‚¬μ©ν•μ§€ μ•μ
    {2, 3, 8},  // 1λ² λ…Έλ“μ— μ—°κ²°λ λ…Έλ“λ“¤
    {1, 7},
    {1, 4, 5},
    {3, 5},
    {3, 4},
    {7},
    {2, 6, 8},
    {1, 7}
};

vector<bool> visited(9, false); // λ°©λ¬Έ κΈ°λ΅

void bfs(int start) {
    queue<int> q;
    q.push(start);
    visited[start] = true;

    while (!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";

        for (int i = 0; i < graph[node].size(); i++) {
            int next = graph[node][i];
            if (!visited[next]) {
                q.push(next);
                visited[next] = true;
            }
        }
    }
}

int main() {
    bfs(1); 
    return 0;
}
